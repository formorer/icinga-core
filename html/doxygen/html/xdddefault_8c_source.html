00101

00104                         save=FALSE;

00112                         save=FALSE;
00113

00115                 if(save==FALSE){
00116                         update_file=TRUE;
00117                         delete_downtime(temp_downtime->type,temp_downtime->downtime_id);
00118                         }
00119                 }
00120

00122         for(temp_downtime=scheduled_downtime_list;temp_downtime!=NULL;temp_downtime=next_downtime){
00123
